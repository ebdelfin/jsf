13   public class AuthFilter implements Filter { 
14       @Override

16

18    
19       @Override 
20       public void doFilter(ServletRequest servletRequest, ServletResponse servletResponse, FilterChain filterChain) throws IOException, ServletException { 
21           // Get the session from session attribute 
22           AuthBean authBean = (AuthBean)((HttpServletRequest)servletRequest) 
23                                   .getSession().getAttribute("authBean"); 
24    
25           // For the first application request there is no session in the session so user needs to log in 
26           // For other requests session is present but we need to check if user has logged in successfully 
27           if (authBean == null ||  
28                   authBean.getLoggedUsername() == null ||  
29                   authBean.getLoggedUsername().isEmpty()) { 
30               String contextPath = ((HttpServletRequest)servletRequest).getContextPath(); 
31               ((HttpServletResponse)servletResponse).sendRedirect(contextPath + "/index.jsf"); 
32           } 
33    
34           filterChain.doFilter(servletRequest, servletResponse); 
35       }
